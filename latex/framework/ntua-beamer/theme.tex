\newcommand{\makepresentationtitle}{%

  \begin{center}
    {\usebeamerfont{title}\color{ntuablue}\inserttitle\par}
    \vspace{0.16cm}
    {\color{ntuagold}\rule{0.44\paperwidth}{0.8pt}\par}
    \vspace{0.22cm}
    {\usebeamerfont{subtitle}\color{ntuatext}\insertsubtitle\par}
    \vspace{0.34cm}
    {\usebeamerfont{author}\color{ntuablue}\insertauthor\par}
    \vspace{0.14cm}
    {\usebeamerfont{date}\insertdate\par}
  \end{center}

  \vfill
}

\setbeamertemplate{frametitle}{%
  \nointerlineskip
  \begin{beamercolorbox}[wd=\paperwidth,ht=1.02cm,dp=0.24cm]{frametitle}
    \raisebox{-0.24cm}[1.02cm][0.24cm]{%
      \vbox to \dimexpr1.02cm+0.24cm\relax{%
        \vfill
        \hbox to \paperwidth{%
          \hspace*{0.85cm}%
          \parbox[c]{\dimexpr\paperwidth-0.85cm-0.55cm-\headerlogoboxwidth-0.24cm\relax}{%
            \raggedright
            \usebeamerfont{frametitle}\strut\insertframetitle\strut
          }%
          \hspace*{0.24cm}%
          \parbox[c]{\headerlogoboxwidth}{%
            \centering
            \insertheaderlogo
          }%
          \hspace*{0.55cm}%
        }%
        \vfill
      }%
    }%
  \end{beamercolorbox}
}

\setbeamertemplate{footline}{%
  \leavevmode
  \ifslidefootrefsemptyTF{%
    \begin{beamercolorbox}[wd=\paperwidth,ht=0.34cm,dp=0.16cm,leftskip=\footlineleftinset,rightskip=\footlinerightinset]{footline}
      \hbox to \dimexpr\paperwidth-\footlineleftinset-\footlinerightinset\relax{%
        \hfill
        \parbox[b]{\footpagenumberwidth}{%
          \raggedleft
          {\usebeamerfont{footline}\color{ntuablue}\insertframenumber/\inserttotalframenumber}%
        }%
      }%
    \end{beamercolorbox}
  }{%
    \begin{beamercolorbox}[wd=\paperwidth,ht=0pt,dp=0pt,leftskip=\footlineleftinset,rightskip=\footlinerightinset,sep=0pt]{footline}
      \raisebox{0pt}[0pt][0pt]{%
        \vbox{%
          \vskip 0.08cm
          \hbox to \dimexpr\paperwidth-\footlineleftinset-\footlinerightinset\relax{%
            \vbox{%
              \hsize=\dimexpr\paperwidth-\footlineleftinset-\footlinerightinset-\footpagenumberwidth\relax
              \raggedright
              \usebeamerfont{footnote}%
              \color{ntuatext}%
              \noindent
              \insertslidefootrefs
            }%
            \hfill
            \vbox{%
              \hsize=\footpagenumberwidth
              \raggedleft
              \usebeamerfont{footline}%
              \color{ntuablue}%
              \noindent
              \insertframenumber/\inserttotalframenumber
            }%
          }%
          \vskip 0.08cm
        }%
      }%
    \end{beamercolorbox}
  }%
}
