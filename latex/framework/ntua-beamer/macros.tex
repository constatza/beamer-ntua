  {
    \prop_if_in:NnTF \g_ntua_slide_refnum_prop {#1}
      {
        \prop_get:NnN \g_ntua_slide_refnum_prop {#1} \l_tmpa_tl
        \ntuafootnotemarknum{\l_tmpa_tl}
      }
      {
        \prop_get:NnNTF \g_ntua_slide_source_prop {#1} \l_tmpa_tl
          {
            \ntua_slide_ref_store:nn {#1} { \exp_not:V \l_tmpa_tl }
          }
          {
            \cs_if_exist:NTF \fullcite
              {
                \ntua_slide_ref_store:nn {#1} { \noexpand\fullcite{#1} }
              }
              {
                \msg_error:nnn {ntua-beamer}{undefined-slide-source}{#1}
              }
          }
      }
  }

    \tl_set:Ne \l_tmpb_tl { \int_eval:n { \value{footnote} } }

      }

  }
