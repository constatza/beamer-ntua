  {
    \tl_if_blank:eTF {#2}
      {
        \tl_if_blank:eTF {#3}
          { \includegraphics{#1} }
          { \includegraphics[keepaspectratio,height=#3]{#1} }
      }
      {
        \tl_if_blank:eTF {#3}
          { \includegraphics[keepaspectratio,width=#2]{#1} }
          { \includegraphics[keepaspectratio,width=#2,height=#3]{#1} }
      }
  }
\cs_new_protected:Npn \ntua_include_named_logo:nnn #1#2#3
  {
    \tl_if_blank:eTF { \ntuabeamerassetpath }
      {
        \IfFileExists{assets/#1}
          { \ntua_include_logo_file:nnn {assets/#1} {#2} {#3} }
          {
            \IfFileExists{#1}
              { \ntua_include_logo_file:nnn {#1} {#2} {#3} }
              { }
          }
      }
      {
        \IfFileExists{\ntuabeamerassetpath/#1}
          { \ntua_include_logo_file:nnn {\ntuabeamerassetpath/#1} {#2} {#3} }
          {
            \IfFileExists{assets/#1}
              { \ntua_include_logo_file:nnn {assets/#1} {#2} {#3} }
              {
                \IfFileExists{#1}
                  { \ntua_include_logo_file:nnn {#1} {#2} {#3} }
                  { }
              }
          }
      }
  }

    \raisebox{-0.24cm}[1.02cm][0.24cm]{%
      \vbox to \dimexpr1.02cm+0.24cm\relax{%
        \vfill
        \hbox to \paperwidth{%
          \hspace*{0.85cm}%

            \centering

          \parbox[c]{\dimexpr\paperwidth-0.85cm-0.55cm-\ntuaheaderslotwidth{\ntuabeamerheaderleftlogo}{\ntuabeamerheaderleftlogomaxwidth}-\ntuaheaderslotgap{\ntuabeamerheaderleftlogo}-\ntuaheaderslotwidth{\ntuabeamerheaderrightlogo}{\ntuabeamerheaderrightlogomaxwidth}-\ntuaheaderslotgap{\ntuabeamerheaderrightlogo}\relax}{%
            \raggedright
            \usebeamerfont{frametitle}\strut\insertframetitle\strut

  }%
