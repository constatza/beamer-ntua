\msg_new:nnn {ntua-beamer}{undefined-frame-source}
  { Frame~source~'#1'~is~undefined. }
\cs_new_protected:Npn \ntua_frame_refs_reset:
  {
    \prop_gclear:N \g_ntua_frame_refnum_prop
    \seq_gclear:N \g_ntua_frame_reflines_seq
    \setcounter{footnote}{0}
  }
\cs_new_protected:Npn \ntua_define_frame_source:nn #1#2
  {
    \prop_gput:Nnn \g_ntua_frame_source_prop {#1} {#2}
  }
\cs_new_protected:Npn \ntua_frame_cite:n #1
  {
    \prop_if_in:NnTF \g_ntua_frame_refnum_prop {#1}
      {
        \prop_get:NnN \g_ntua_frame_refnum_prop {#1} \l_tmpa_tl
        \ntuafootnotemarknum{\l_tmpa_tl}
      }
      {
        \prop_get:NnNTF \g_ntua_frame_source_prop {#1} \l_tmpa_tl
          {
            \ntua_frame_ref_store:nn {#1} { \exp_not:V \l_tmpa_tl }
          }
          {
            \cs_if_exist:NTF \fullcite
              {
                \ntua_frame_ref_store:nn {#1} { \exp_not:n { \fullcite{#1} } }
              }
              {
                \msg_error:nnn {ntua-beamer}{undefined-frame-source}{#1}
              }
          }
      }
  }
\cs_new_protected:Npn \ntua_frame_ref_store:nn #1#2
  {
    \stepcounter{footnote}
    \tl_set:Ne \l_tmpb_tl { \int_eval:n { \value{footnote} } }
    \prop_gput:NnV \g_ntua_frame_refnum_prop {#1} \l_tmpb_tl
    \seq_gput_right:Ne \g_ntua_frame_reflines_seq
      {
        \exp_not:N \ntuafootnotemarknum { \tl_use:N \l_tmpb_tl }
        ~
        #2
      }
    \ntuafootnotemarknum{\l_tmpb_tl}
  }

    \seq_use:Nn \g_ntua_frame_reflines_seq { \enspace \textbar \enspace }
  }
